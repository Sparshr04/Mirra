# ============================================================
# AMD_Frontend_2 — npm Package Requirements
# ============================================================
# This is a React (JavaScript) project.
# There is NO Python requirements.txt — dependencies are
# managed via npm using package.json.
#
# ──────────────────────────────────────────────────────────
#  QUICKSTART (install everything in one command):
# ──────────────────────────────────────────────────────────
#
#   npm install
#
# ──────────────────────────────────────────────────────────
#  OR install packages individually (see list below):
# ──────────────────────────────────────────────────────────
#
# NODE VERSION REQUIREMENT
#   Node.js >= 16.x
#   npm     >= 8.x
#   Install: https://nodejs.org
#
# ============================================================
# PRODUCTION DEPENDENCIES
# ============================================================

# ── Core React ──────────────────────────────────────────────
# npm install react@^18.2.0
react@^18.2.0

# npm install react-dom@^18.2.0
react-dom@^18.2.0

# npm install react-scripts@5.0.1
react-scripts@5.0.1

# ── 3D Rendering (Three.js Stack) ───────────────────────────
# Used in: ViewerPage.jsx — renders PLY point clouds in WebGL

# npm install three@^0.160.0
three@^0.160.0

# npm install @react-three/fiber@^8.15.16
@react-three/fiber@^8.15.16

# npm install @react-three/drei@^9.105.6
# Provides: OrbitControls, Points, PointMaterial, ContactShadows, Grid
@react-three/drei@^9.105.6

# npm install three-stdlib@^2.36.1
# Provides: PLYLoader (used to load .ply files from the backend)
three-stdlib@^2.36.1

# ── Spline 3D Scene ─────────────────────────────────────────
# Used in: HeroPage.jsx — interactive 3D hero animation

# npm install @splinetool/react-spline@^4.1.0
@splinetool/react-spline@^4.1.0

# npm install @splinetool/runtime@^1.12.61
@splinetool/runtime@^1.12.61

# ── Performance Monitoring ───────────────────────────────────
# npm install web-vitals@^2.1.4
web-vitals@^2.1.4

# ============================================================
# DEV DEPENDENCIES
# ============================================================

# ── CSS Utilities ────────────────────────────────────────────
# npm install --save-dev tailwindcss@^3.2.7
tailwindcss@^3.2.7

# npm install --save-dev stylelint@^17.3.0
stylelint@^17.3.0

# npm install --save-dev stylelint-config-standard@^40.0.0
stylelint-config-standard@^40.0.0

# ============================================================
# FULL INSTALL COMMANDS (copy-paste into terminal)
# ============================================================
#
# Step 1: Install all production + dev deps at once
#   npm install
#
# Step 2: Start dev server
#   npm start
#
# Step 3: Build for production
#   npm run build
#
# ============================================================
# NOTES
# ============================================================
#
# • @react-three/drei requires three and @react-three/fiber as peers
# • three-stdlib must match your three version (0.160.x → use drei ^9)
# • @splinetool/react-spline requires internet access to load the
#   Spline runtime from CDN on first load
# • The backend (AMD_Backend) must be running on port 8000 for
#   upload/processing/viewer features to work
#   → cd AMD_Backend && uvicorn src.api.server:app --port 8000 --reload
